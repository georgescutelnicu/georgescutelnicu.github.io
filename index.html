class Education:
    def __init__(self, degree):
        self.degree = degree
        self.certificates = {}

    def add_certificate(self, name, issuer, year, view_certificate):
        self.certificates[name] = {
            "Issuer": issuer,
            "Certificate Year": year,
            "View Certificate": view_certificate
        }
